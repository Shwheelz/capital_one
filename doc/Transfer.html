def self.createTransfer(toAcc, json)
        url = "#{self.urlWithEntity}/#{toAcc}/Transfers?key=#{self.apiKey}"
        uri = URI.parse(url)
        http = Net::HTTP.new(uri.host, uri.port)
        request = Net::HTTP::Post.new(uri.request_uri, initheader = {'Content-Type' => 'application/json'})
        request.body = json
        resp = http.request(request)
        data = JSON.parse(resp.body)
end

    Deletes a specified Transfer from a specified account. Parameters: AccountID, TransferID Returns http response code.

    Note:¶ ↑

    This does not actually delete the Transfer from the database, it only sets it's status to 'cancelled'

    # File lib/capital_one/transfer.rb, line 68
def self.deleteTransfer(accID, transID)
        url = "#{self.urlWithEntity}/#{accID}/Transfers/#{transID}?key=#{self.apiKey}"
        uri = URI.parse(url)
        http = Net::HTTP.new(uri.host, uri.port)
        key="?key=#{self.apiKey}"
        request = Net::HTTP::Delete.new(uri.path+key)
        resp = http.request(request)
        data = JSON.parse(resp.body)
end
